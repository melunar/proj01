画布canvas绘制工作必须在 JavaScript 内部完成

渐变，贴图
浏览器不支持画布。
基本方法

var c = document.getElementById("id");
var cxt = c.getContext("2d");

填充形状
cxt.fillStyle="#FF0000";
cxt.fillRect(0,0,150,75);
画线
cxt.moveTo(10,10);
cxt.lineTo(150,50);
cxt.lineTo(10,50);
cxt.stroke();//生成
画圆形
cxt.fillStyle="#FF0000";
cxt.beginPath();
cxt.arc(70,18,15,0,Math.PI*2,true);
cxt.closePath();
cxt.fill();
贴图
var img=new Image() img.src="img.png" cxt.drawImage(img,0,0);
